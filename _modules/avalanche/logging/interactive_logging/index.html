    @property
    def _progress(self):
        if self._pbar is None:
            self._pbar = tqdm(leave=True, position=0, file=sys.stdout)
        return self._pbar


[docs]    def _end_progress(self):
        if self._pbar is not None:
            self._pbar.close()
            self._pbar = None
